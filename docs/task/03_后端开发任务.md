### B2-1：Session 管理器
**文件**：`app/core/session.py`

```python
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Literal
import uuid

class SessionStatus(Enum):
    RUNNING = "running"
    PAUSED = "paused"       # 等待人工干预
    DONE = "done"
    ERROR = "error"

@dataclass
class Session:
    session_id: str = field(default_factory=lambda: str(uuid.uuid4()))
    story_id: str = ""
    created_at: datetime = field(default_factory=datetime.now)
    status: SessionStatus = SessionStatus.RUNNING
    current_agent: str | None = None
    mode: Literal["auto", "human"] = "auto"

class SessionManager:
    """全局 Session 管理器"""

    def __init__(self, max_concurrent: int = 3):
        self._sessions: dict[str, Session] = {}
        self._semaphore = asyncio.Semaphore(max_concurrent)

    async def create_session(self, story_id: str, mode: str) -> Session:
        session = Session(story_id=story_id, mode=mode)
        self._sessions[session.session_id] = session
        return session

    async def get_session(self, session_id: str) -> Session | None:
        return self._sessions.get(session_id)

    async def update_status(self, session_id: str, status: SessionStatus, current_agent: str = None):
        if session_id in self._sessions:
            self._sessions[session_id].status = status
            self._sessions[session_id].current_agent = current_agent
```

**要求**：
- Session 状态可持久化到 Redis（支持服务重启后恢复）
- 并发数通过 Semaphore 控制（`max_concurrent` 可配置）

### B2-2：三层记忆系统
**文件**：`app/core/memory/__init__.py`, `short_term.py`, `medium_term.py`, `long_term.py`

```python
# app/core/memory/__init__.py
from .short_term import ShortTermMemory
from .medium_term import MediumTermMemory
from .long_term import LongTermMemory

class MemoryLayer:
    """三层记忆聚合接口"""

    def __init__(self, session: Session):
        self.session = session
        self.short = ShortTermMemory()
        self.medium = MediumTermMemory(session)
        self.long = LongTermMemory()

    async def search(self, query: str, top_k: int = 3) -> list[dict]:
        """检索长期记忆，返回 top_k 条相似记忆"""
        return await self.long.search(
            session_id=self.session.session_id,
            query=query,
            top_k=top_k
        )

    async def add(self, agent_name: str, output: dict):
        """写入中期 + 长期记忆"""
        await self.medium.append(agent_name, output)
        # 向量化写入长期记忆
        await self.long.add(
            session_id=self.session.session_id,
            agent_name=agent_name,
            content=output
        )

    async def get_context_for_agent(self, agent_name: str) -> dict:
        """为某个 Agent 准备完整的上下文"""
        medium = await self.medium.get_history(agent_name)
        return {"medium_term": medium}
```

**短期记忆**（`short_term.py`）：
- 基于 Python `deque` 或列表
- 保存 Agent 消息历史（user/assistant/tool_result）
- 每个 `Agent.execute()` 调用前传入，调用后更新
- 最大长度可配置（默认 50 条消息）

**中期记忆**（`medium_term.py`）：
- 每个 Session 一个 JSON 文件：`data/memory/sessions/{session_id}.json`
- 结构：
  ```json
  {
    "session_id": "...",
    "story_id": "...",
    "history": [
      { "agent": "ScriptAnalysis", "output": {...}, "timestamp": "..." },
      { "agent": "PlotDeconstruct", "output": {...}, "timestamp": "..." }
    ]
  }
  ```

**长期记忆**（`long_term.py`）：
- 基于 ChromaDB（默认，轻量）
- 字段：session_id, agent_name, content（文本）, embedding, timestamp
- 查询：向量相似度检索，支持 `top_k`
- 可切换 Qdrant / Milvus（通过 `config.py` 配置）

### B2-3：pi-mono Agent 基类
**文件**：`app/agents/base.py`

```python
from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Any
import json

@dataclass
class AgentContext:
    """Agent 执行时的完整上下文"""
    session_id: str
    story_id: str
    input_data: dict          # 来自上一个 Agent 或初始输入
    memory_context: list[dict] # 长期记忆检索结果
    human_feedback: str | None = None
    session: "Session" = None  # 引用 Session（可调用 memory.add）

class BaseAgent(ABC):
    """pi-mono Agent 基类——所有 Pipeline Agent 的父类"""

    name: str = "BaseAgent"
    description: str = ""

    def __init__(self, session: "Session"):
        self.session = session

    async def execute(self, context: AgentContext) -> dict:
        """
        统一的 execute 流程：
        1. 检索相关记忆（memory.search）
        2. 构建 prompt（注入记忆上下文）
        3. 调用 LLM
        4. 解析输出
        5. 写入记忆（memory.add）
        6. 返回输出
        """
        # 1. 检索记忆
        query = self._get_search_query(context.input_data)
        memory_results = await self.session.memory.search(query, top_k=3)

        # 2. 构建 prompt
        prompt = self._build_prompt(context, memory_results)

        # 3. LLM 调用
        output = await self._call_llm(prompt, context.human_feedback)

        # 4. 解析输出
        parsed = self._parse_output(output)

        # 5. 写入记忆
        await self.session.memory.add(self.name, parsed)

        return parsed

    @abstractmethod
    def _get_search_query(self, input_data: dict) -> str:
        """从输入数据中提取用于检索记忆的 query"""
        raise NotImplementedError

    @abstractmethod
    def _build_prompt(self, context: AgentContext, memory_results: list[dict]) -> str:
        """构建发送给 LLM 的完整 prompt"""
        raise NotImplementedError

    @abstractmethod
    async def _call_llm(self, prompt: str, human_feedback: str | None) -> str:
        """调用 LLM，返回原始文本输出"""
        raise NotImplementedError

    @abstractmethod
    def _parse_output(self, raw_output: str) -> dict:
        """解析 LLM 输出为结构化 dict"""
        raise NotImplementedError
```

**关键设计**：
- 所有 Agent 继承 `BaseAgent`，只需实现 4 个抽象方法
- `human_feedback` 非空时，prompt 末尾追加反馈内容，重新执行
- 记忆检索在 `execute()` 入口统一处理，子类只关注业务逻辑

```python
from typing import Callable, Any

class ToolRegistry:
    """统一工具注册表——所有 Agent 共享工具集"""

    def __init__(self):
        self._tools: dict[str, Callable] = {}

    def register(self, name: str, func: Callable):
        self._tools[name] = func

    def get(self, name: str) -> Callable | None:
        return self._tools.get(name)

    def list_tools(self) -> list[str]:
        return list(self._tools.keys())
```

## B3：Pipeline Orchestrator（编排器）

**文件**：`app/core/orchestrator.py`

**核心逻辑**：
```python
class PipelineOrchestrator:
    """多 Agent 协作编排器"""

    def __init__(self, session: Session, mode: Literal["auto", "human"]):
        self.session = session
        self.mode = mode
        self.agents = PIPELINE_AGENTS  # [ScriptAnalysisAgent, ...]

    async def run(self, story_id: str, initial_input: dict, ws: WebSocket):
        self.session.status = SessionStatus.RUNNING

        current_input = initial_input

        for agent_cls in self.agents:
            agent = agent_cls(session=self.session)
            self.session.current_agent = agent.name

            # 通知前端：Agent 开始
            await ws.send_json({
                "type": "agent_status",
                "agent": agent.name,
                "status": "running"
            })

            # 构建上下文（包含记忆）
            context = AgentContext(
                session_id=self.session.session_id,
                story_id=story_id,
                input_data=current_input,
                memory_context=[],
                session=self.session
            )

            # 执行 Agent
            output = await agent.execute(context)

            # 通知前端：Agent 完成
            await ws.send_json({
                "type": "agent_output",
                "agent": agent.name,
                "output": output
            })

            # 人工干预模式
            if self.mode == "human":
                self.session.status = SessionStatus.PAUSED
                await ws.send_json({
                    "type": "agent_waiting",
                    "agent": agent.name,
                    "output": output
                })
                intervention = await self._wait_for_intervention(story_id, agent.name)

                if intervention["action"] == "regenerate":
                    context.human_feedback = intervention["feedback"]
                    output = await agent.execute(context)
                elif intervention["action"] == "skip":
                    output = DEFAULT_OUTPUTS.get(agent.name, {})

                self.session.status = SessionStatus.RUNNING

            current_input = output  # 传入下一个 Agent

        self.session.status = SessionStatus.DONE
        await ws.send_json({"type": "story_complete", "final_output": current_input})

    async def _wait_for_intervention(self, story_id: str, agent_name: str) -> dict:
```
